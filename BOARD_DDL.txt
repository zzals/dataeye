DROP TABLE PEN_BOARD;
CREATE TABLE PEN_BOARD (
  BOARD_NO NUMBER NOT NULL PRIMARY KEY,
  BOARD_DIV VARCHAR2(100) NOT NULL,
  TITLE VARCHAR2(1000) NOT NULL,
  CONT CLOB  NOT NULL,
  SDATE TIMESTAMP,
  EDATE TIMESTAMP,
  READ_CNT NUMBER,
  REG_ID VARCHAR2(200),
  REG_DATE TIMESTAMP
);

DROP TABLE PEN_BOARD_REPLY;
CREATE TABLE PEN_BOARD_REPLY (
  REPLY_NO  NUMBER PRIMARY KEY,
  BOARD_NO NUMBER NOT NULL, 
  BOARD_DIV VARCHAR2(100) NOT NULL,
  CONT CLOB  NOT NULL,  
  REG_ID VARCHAR2(200),
  REG_DATE TIMESTAMP
);

DROP TABLE PEN_BOARD_FILE;
CREATE TABLE PEN_BOARD_FILE (
  FILE_NO NUMBER NOT NULL PRIMARY KEY,
  BOARD_NO NUMBER NOT NULL,
  BOARD_DIV VARCHAR2(100) NOT NULL,
  PATH VARCHAR2(1000) NOT NULL,
  ORG_NAME VARCHAR2(1000) NOT NULL,
  SYS_NAME VARCHAR2(1000) NOT NULL,
  FILE_SIZE NUMBER,
  DOWN_CNT NUMBER
);
